	for (int y = 0; y < 7; y++)

int main()
{
	setlocale(0, "");

	int nachalo[] = { 6,2,2,5,7,3,5,1,4,6,2,4 };
	int kollvo[] = { 31,28,31,30,31,30,31,31,30,31,30,31 };
	int t, k = 0, mas[12][6][7];
	int temp[42] = { 0 };

	for (int m = 0; m < 12; m++)
	{
		t = nachalo[m];
		//Заполнение побочного массива
		for (int i = 0; i < kollvo[m]; i++)
		{
			temp[t + i - 1] = i + 1;
		}

		//Заполнение календаря
		for (int i = 0; i < 6; i++)
		{
			for (int j = 0; j < 7; j++)
			{
				mas[m][i][j] = temp[k];
				k++;
			}
		}

		//Очистка побочного массива
		for (int i = 0; i < 42; i++)
		{
			temp[i] = 0;
		} 
		k = 0;
	}

	printmas(0, 3, mas);
	printmas(3, 6, mas);
	printmas(6, 9, mas);
	printmas(9, 12, mas);
	system("pause");
}
