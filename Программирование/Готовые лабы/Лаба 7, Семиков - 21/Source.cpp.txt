#include "Header.h"

void sort_i_zapis(FOOTBALL fut[N]) {


	char* name = new char[N];
	int* Num = new int[N];

	for (int i = 0; i < N; i++) {
		Num[i] = i;
		name[i] = (char)(fut[i].nam[0]);
	}
	int k = 0; int c = 0;

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < N; j++) {
			if ((char)(name[j]) > (char)(name[j + 1])) {
				k = name[j];
				name[j] = name[j + 1];
				name[j + 1] = k;

				c = Num[j];
				Num[j] = Num[j + 1];
				Num[j + 1] = c;
			}
		}
	}
	
	//==============================

	std::ofstream fout;          // поток для записи
	fout.open("D:\\notex.txt"); // окрываем файл для записи

	if (fout.is_open())
	{
		fout << "\t\tФутбольная команда\n" << std::endl;
		fout << "Название          Лига      Колличество\n" << std::endl;
		for (int i = 0; i < N; i++) {
			fout << " " << fut[Num[i]].nam << "       " << fut[Num[i]].league << "\t" << fut[Num[i]].kollvo << std::endl;
		}
	}
	fout.close();

	//==============================

	std::string line;

	std::ifstream in("D:\\notex.txt"); // окрываем файл для чтения
	if (in.is_open())
	{
		while (getline(in, line))
		{
			std::cout << line << std::endl;
		}
	}
	in.close();     // закрываем файл

	delete[] name; delete[] Num;
}
