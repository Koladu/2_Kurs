#include "Tee.h"

int n = 0; // Количество продуктов
string FailName = "tee.db";  // Путь к файлу и/или имя файла

int main()
{
	system("title Ведомость по закупкам чая");	 //Заголовок
	setlocale(LC_ALL, "ru");

	WriteBeginText();
	ifstream file;	// Чтение файла
	file.open(FailName);  // Открытие файла

	if (file.is_open() == true)	// Если файл открыт можно работать программой
	{
		file.close();	// Закрытие файла
		n = number();
		Tea* tee = new Tea[n];
		Initialization(n, tee);
		menu(n, tee);

		//Draw(n, tee);


		delete[] tee;  //удаление динамической памяти
		_getch();  // Задержка программы
	}
	else{
		cout << "\t ------------------" << "\n\t | Файл не открыт | \n" << "\t ------------------";
		Sleep(2000);  // Если файл не открывается программа выключится через 2 секунды.
	}
	return 0;
}
